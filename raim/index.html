from flask import Flask, render_template, request app = Flask(__name__) # Lista akcji z adnotacjami actions = [] # Strona główna @app.route('/') def index(): return render_template('index.html', actions=actions) # Dodawanie adnotacji @app.route('/add_annotation', methods=['POST']) def add_annotation(): action = request.form['action'] time = request.form['time'] actions.append({'action': action, 'time': time}) return 'Adnotacja dodana' if __name__ == '__main__': app.run(debug=True)
